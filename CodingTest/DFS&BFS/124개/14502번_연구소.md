[문제](https://www.acmicpc.net/problem/14502)

배울 점 : `DFS`를 가지고 `모든 경우의 수` 구하기 

``` java
import java.io.IOException;

import java.io.BufferedReader;

import java.io.InputStreamReader;

import java.util.LinkedList;
import java.util.Queue;
import java.util.StringTokenizer;

class XY {
    int x;
    int y;

    XY(int x, int y) {
        this.x = x;
        this.y = y;
    }
}

public class Main {

    static int count = Integer.MIN_VALUE;
    static int N;
    static int M;
    static int[][] graph;

    public static void main(String[] args) throws IOException {

        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        StringTokenizer st = new StringTokenizer(br.readLine());

        N = Integer.parseInt(st.nextToken()); // 세로 길이
        M = Integer.parseInt(st.nextToken()); // 가로 길이
        
        // 그래프 설정
        graph = new int[N][M];

        for(int i = 0; i < N; i++) {
            st = new StringTokenizer(br.readLine());
            for(int j = 0; j < M; j++) {
                graph[i][j] = Integer.parseInt(st.nextToken());
            }
        }

        dfs(0);

        System.out.println(count);

    }

    // 벽의 개수를 세기 위해서 dfs를 이용한다.
    static void dfs (int wallCount) {

        // 벽을 3개 다 설치했다면
        // 바이러스가 어떻게 퍼지는지 살펴보자.
        if(wallCount == 3) {
            bfs();
            return;
        }

        // 벽을 하나씩 하나씩 세운다.
        for(int i = 0; i < N; i++) {
            for(int j = 0; j < M; j++) {
                if(graph[i][j] == 0) {
                    graph[i][j] = 1;
                    dfs(wallCount + 1);
                    graph[i][j] = 0;
                }
            }
        }
    }

    static void bfs() {

        // 초기 큐 설정
        // 바이러스에 해당하는 모든 좌표 큐에 넣고 시작
        Queue<XY> que = new LinkedList<>();

        for(int i = 0; i < N; i++) {
            for(int j = 0; j < M; j++) {
                if(graph[i][j] == 2) {
                    que.add(new XY(i, j));
                }
            }
        }

        // 복사본
        int[][] copyGraph = new int[N][M];

        for(int i = 0; i < N; i++) {
            copyGraph[i] = graph[i].clone();
        }

        // BFS 탐색
        // 바이러스 확산 시작
        int[] dx = {0, 0, 1, -1}; // 상하좌우
        int[] dy = {1, -1, 0, 0}; // 상하좌우

        while(!que.isEmpty()) {

            XY now = que.poll();

            for(int k = 0; k < 4; k++) {

                int nx = now.x + dx[k];
                int ny = now.y + dy[k];

                if(nx >= 0 && nx < N && ny >= 0 && ny < M) {
                    if(copyGraph[nx][ny] == 0) {
                        copyGraph[nx][ny] = 2;
                        que.add(new XY(nx, ny));
                    }
                }
            }
        }

        findSafe(copyGraph);

    }

    static void findSafe(int[][] copyGraph) {
        int safe = 0;

        for(int i = 0; i < N; i++) {
            for(int j = 0; j < M; j++) {
                if(copyGraph[i][j] == 0) {
                    safe++;
                }
            }
        }

        count = Math.max(safe, count); 
    }



}
```
