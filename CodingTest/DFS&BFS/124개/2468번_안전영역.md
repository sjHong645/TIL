[문제](https://www.acmicpc.net/problem/2468)

비가 얼마나 올 지 정해지지 않았으니까 

비가 `아예 안 올 때` 부터 비가 `건물 최대 높이(maxHeight)만큼` 오는 경우까지 전부 따져봐야 한다. 

각각의 경우에서 만들어지는 `영역의 개수`를 구하는 문제 


``` java
import java.io.IOException;

import java.io.BufferedReader;

import java.io.InputStreamReader;

import java.util.LinkedList;
import java.util.Queue;
import java.util.StringTokenizer;

class XY {
    int x;
    int y;

    XY(int x, int y) {
        this.x = x;
        this.y = y;
    }
}

public class Main {

    static int maxHeight = 0;
    static int maxFieldCount = 0; // 최대 영역 개수

    public static void main(String[] args) throws IOException {

        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st;

        // 그래프 설정
        int N = Integer.parseInt(br.readLine());

        int[][] field = new int[N][N];

        for(int i = 0; i < N; i++) {
            st = new StringTokenizer(br.readLine());
            for(int j = 0; j < N; j++) {
                field[i][j] = Integer.parseInt(st.nextToken());
                maxHeight = Math.max(maxHeight, field[i][j]);
            }
        }

        // 비가 올 때 물에 잠기지 않는 안전한 영역이 최대 몇 개가 될까
        // 여기서 영역은 상하좌우로 이웃한다.

        boolean[][] visited;

        for(int h = 0; h < maxHeight; h++) {
            // bfs(field, h); // 비가 높이 i만큼 내렸을 때 잠기지 않는 영역을 구한다.

            // 비가 h만큼 내렸을 때 잠긴 영역을 표시한다.
            // 잠긴 영역 = 벽 과 같은 의미 
            visited = new boolean[N][N];

            for(int i = 0; i < N; i++) {
                for(int j = 0; j < N; j++) {
                    if(field[i][j] <= h) visited[i][j] = true;
                }
            }

            int temp_maxFieldCount = 0;
            // 다 표시하고 나서 잠기지 않은 영역의 개수를 구한다.
            
            for(int i = 0; i < N; i++) {
                for(int j = 0; j < N; j++) {
                    if(visited[i][j] == false) {
                        dfs(visited, i, j);
                        temp_maxFieldCount++;
                    }
                }
            }

            maxFieldCount = Math.max(temp_maxFieldCount, maxFieldCount);

        }

        System.out.print(maxFieldCount);

    }

    // 잠기지 않은 영역의 개수를 구하는 메소드 
    static void dfs(boolean[][] visited, int i, int j) {

        visited[i][j] = true;
        int[] dx = {0, 0, -1, 1}; // 상하좌우
        int[] dy = {-1, 1, 0, 0}; // 상하좌우

        for(int k = 0; k < 4; k++) {
            int nx = i + dx[k];
            int ny = j + dy[k];

            if(nx >= 0 && nx < visited.length && ny >= 0 && ny < visited.length
            && visited[nx][ny] == false) {
              dfs(visited, nx, ny);
            }
        }
    }
}
```
