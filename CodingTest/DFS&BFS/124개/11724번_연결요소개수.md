[문제](https://www.acmicpc.net/problem/11724)

주어진 정보를 바탕으로 `독립된 그래프의 개수`가 `몇 개`인지 파악하는 문제

``` java
import java.io.IOException;

import java.io.BufferedReader;

import java.io.InputStreamReader;

import java.util.StringTokenizer;

public class yes {

    static int count = 1;

    public static void main(String[] args) throws IOException {

        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        StringTokenizer st = new StringTokenizer(br.readLine());

        int N = Integer.parseInt(st.nextToken()); // 정점의 개수
        int M = Integer.parseInt(st.nextToken()); // 간선의 개수

        // 주어진 정보를 가지고 그래프를 만들었을 때
        // 몇 개의 독립적인 그래프가 나오는가

        // 그래프 설정
        int[][] graph = new int[N+1][N+1];
        boolean[] visited = new boolean[N+1];

        for(int i = 0; i < M; i++) {

            st = new StringTokenizer(br.readLine());

            int from = Integer.parseInt(st.nextToken());
            int to = Integer.parseInt(st.nextToken());

            graph[from][to] = 1; graph[to][from] = 1;

        }

        // 연결 요소 파악
        int result = 0;

        for(int i = 1; i <= N; i++) {
            if(visited[i] == false) {
                dfs(graph, visited, i);
                result++;
            }
        }

        System.out.println(result);

    }

    static void dfs(int[][] graph, boolean[] visited, int index) {

        if(visited[index] == true) return;

        visited[index] = true;
        for(int i = 1; i < graph.length; i++) {
            if(graph[index][i] == 1) dfs(graph, visited, i);
        }
    }

}
```
