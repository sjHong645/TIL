[문제](https://www.acmicpc.net/problem/2644)

주어진 두 점의 `최단 경로` 구하는 문제

DFS를 이용했음

``` java
import java.io.IOException;

import java.io.BufferedReader;

import java.io.InputStreamReader;

import java.util.StringTokenizer;


public class Main {

    static int[][] graph;
    static boolean[][] visited;
    static int ans = -1;

    public static void main(String[] args) throws IOException {

        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st;

        // 전체 사람의 수
        int N = Integer.parseInt(br.readLine());
        graph = new int[N+1][N+1];
        visited = new boolean[N+1][N+1];

        // A와 B간에 촌수 구하기
        st = new StringTokenizer(br.readLine());

        int A = Integer.parseInt(st.nextToken());
        int B = Integer.parseInt(st.nextToken());

        // 부모와 자식간의 관계 개수
        int rel = Integer.parseInt(br.readLine());

        for(int i = 0; i < rel; i++) {
            st = new StringTokenizer(br.readLine());
            int m = Integer.parseInt(st.nextToken());
            int n = Integer.parseInt(st.nextToken());

            graph[m][n] = 1; graph[n][m] = 1;

        }


        // 결국에는 A -> B 까지의 최단 경로 구하기
        dfs(A, B, 0);

        System.out.print(ans);

    }

    static void dfs(int start, int end, int dist) {

        if(start == end) {
            ans = dist;
            return;
        }

        for(int i = 0; i < graph[start].length; i++) {
            if(!visited[start][i] && graph[start][i] == 1) {
            
                // start와 이웃한 노드에 대해서 
                // 방문했다는 걸 표시한다.
                visited[start][i] = true;
                visited[i][start] = true;
                
                // 이웃한 노드부터 다시 시작한다.
                dfs(i, end, dist + 1);
            }
        }
    }
}
```
