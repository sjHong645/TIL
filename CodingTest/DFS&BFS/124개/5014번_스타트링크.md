[문제](https://www.acmicpc.net/problem/5014)

문제가 어렵지는 않았다. 

다만, `예외 처리`를 하지 않아서 제대로 풀지 못했다. 

``` java
import java.io.IOException;
import java.io.BufferedReader;
import java.io.InputStreamReader;

import java.util.LinkedList;
import java.util.Queue;
import java.util.StringTokenizer;

class Node{
    int pos;
    int count;

    Node(int pos, int count) {
        this.pos = pos;
        this.count = count;
    }
}

public class yes {

    public static void main(String[] args) throws IOException {

        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st;

        // 그래프 설정
        st = new StringTokenizer(br.readLine());

        int F, S, G, U, D;

        F = Integer.parseInt(st.nextToken()); // 총 건물 층수
        S = Integer.parseInt(st.nextToken()); // 지금 있는 위치
        G = Integer.parseInt(st.nextToken()); // 사무실 위치
        U = Integer.parseInt(st.nextToken()); // 한 번 올라가는 층수
        D = Integer.parseInt(st.nextToken()); // 한 번 내려가는 층수

        // 지금 있는 층이 
        // 사무실 층이라면 움직일 필요 없다. 
        if(S == G) {
            System.out.print(0);
            return;
        }

        boolean[] visited = new boolean[F+1];

        Queue<Node> que = new LinkedList<>();
        que.offer(new Node(S, 0));
        visited[S] = true;

        int count = Integer.MAX_VALUE;
        while(!que.isEmpty()) {

            Node now = que.poll();

            // 사무실이 있는 곳에 왔다면 count를 갱신
            if(now.pos + U == G || now.pos - D == G) {
                count = Math.min(now.count + 1, count);
                continue;
            }
            
            // 위층으로 이동했을 때 
            if(now.pos + U <= F && !visited[now.pos + U]) {
                visited[now.pos + U] = true;
                que.offer(new Node(now.pos + U, now.count+1));
            }

            // 아래층으로 이동했을 때 
            if(now.pos - D > 0 && !visited[now.pos - D]) {
                visited[now.pos - D] = true;
                que.offer(new Node(now.pos - D, now.count+1));
            }

        }

        System.out.print(count == Integer.MAX_VALUE ? "use the stairs" : count);

    }
}
```
