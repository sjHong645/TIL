[문제](https://leetcode.com/problems/set-mismatch/)

## 정답 코드 
``` java
class Solution {
    public int[] findErrorNums(int[] nums) {

        // 중복된 숫자를 찾아서 
        // 그 숫자와 
        // 원래 그 숫자에 있어야할 값을 배열로 return 
        
        int[] result = new int[2]; 

        // result[0] = 중복된 숫자
        // result[1] = 잃어버린 숫자 
        
        // 편의를 위해 0번째는 버린다. 
        boolean[] visited = new boolean[nums.length + 1]; 

        for(int i = 0; i < nums.length; i++) {
            
            // 이미 nums[i]가 존재했다면 
            // 중복된 숫자라는 뜻이니까 result[0]의 값을 설정한다. 
            if(visited[nums[i]] == true) result[0] = nums[i]; 

            visited[nums[i]] = true; 
            
        }
        
        // visited[i] == false라는 건
        // i라는 숫자가 등장한 적이 없다는 뜻이다. 
        
        // 즉, 원래 있어야 할 숫자이므로 
        // 해당 값은 result[1]에 저장한다. 
        for(int i = 1; i < visited.length; i++) {
            if(visited[i] == false) {
                result[1] = i; 
                break; 
            }
        }

        return result;
        
    }
}
```
