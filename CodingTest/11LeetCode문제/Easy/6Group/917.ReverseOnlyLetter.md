[문제](https://leetcode.com/problems/reverse-only-letters/description/)

## 정답 코드 
``` java
class Solution {
    public String reverseOnlyLetters(String s) {
        
        // s에서 알파벳만 따로 모아서 뒤집는다. 
        StringBuilder sb = new StringBuilder(); 

        for(int i = 0; i < s.length(); i++) {
            char ith = s.charAt(i); 
            if(Character.isLetter(ith)) sb.append(ith); 
        }

        String rev = sb.reverse().toString(); 

        // 뒤집은 문자열을 초기화 저장했으니 sb를 초기화
        sb.setLength(0); 
        
        // i = 원래 문자열 s의 인덱스
        // j = 뒤집은 문자열 rev의 인덱스 
        int i = 0, j = 0; 
        while(i < s.length() && j < rev.length()) {
        
            // s의 i번째 인덱스에 알파벳이 있다면 
            // rev의 알파벳을 넣고 j를 1 증가시킨다. 
            if(Character.isLetter(s.charAt(i))) {
                sb.append(rev.charAt(j++)); 
            }
            
            
            // 알파벳이 아니라면 s의 i번째 문자를 그대로 집어넣는다. 
            else sb.append(s.charAt(i)); 

            i++;
        }
        
        // 알파벳은 전부 저장했는데 
        // 그 이외의 기호들을 넣지 못하는 상황이 생길 수 있다. 
        
        // 나머지 기호들을 전부 저장한다. 
        while(i < s.length()) {
            sb.append(s.charAt(i++)); 
        }

        return sb.toString(); 
        
    }
}
```
