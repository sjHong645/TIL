[문제](https://leetcode.com/problems/find-winner-on-a-tic-tac-toe-game/description/)

## 정답 코드 
``` java
class Solution {
    public String tictactoe(int[][] moves) {
        
        // a와 b가 차지한 각 행과 열
        // ex) aRow[2] = 2 -> a가 2번째 행에 표시한 'X' 문구의 개수는 2개 
        
        int[] aRow = new int[3], aCol = new int[3]; 
        int[] bRow = new int[3], bCol = new int[3];

        // a와 b가 차지한 대각선 
        // 왼쪽 => 오른쪽으로 향하는 대각선
        // 오른쪽 => 왼쪽으로 향하는 대각선 
        int aD1 = 0, aD2 = 0, bD1 = 0, bD2 = 0;

        for (int i = 0; i < moves.length; ++i) {

            int r = moves[i][0], c = moves[i][1];
            
            // B 차례 
            if (i % 2 == 1) {
                if (++bRow[r] == 3 || ++bCol[c] == 3 || r == c && ++bD1 == 3 || r + c == 2 && ++bD2 == 3) return "B";
            }
            // A 차례 
            else {
                if (++aRow[r] == 3 || ++aCol[c] == 3 || r == c && ++aD1 == 3 || r + c == 2 && ++aD2 == 3) return "A";
            }
        }
        
        // moves를 완료했는데 
        // 9칸을 다 채웠다면 끝난거니까 Draw
        // 아니면, 아직 안 끝난거니까 Pending 
        return moves.length == 9 ? "Draw" : "Pending";        
    }
}
```
